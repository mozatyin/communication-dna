```python
# ── Detector output bias correction ──────────────────────────────────────────
# Corrections for features with consistent systematic over/under-estimation.
# Values determined from v0.9 + v1.0 eval data (mean direction across all profiles).
# Only applied when direction is consistent across all measured profiles.

DETECTOR_BIAS_CORRECTION: dict[str, float] = {
    "metacommentary": -0.10,
    "vulnerability_willingness": -0.08,
    "disclosure_depth": -0.08,
}


def _apply_detector_bias(features: list[Feature]) -> list[Feature]:
    """Apply systematic bias correction to detector output."""
    corrected = []
    for f in features:
        offset = DETECTOR_BIAS_CORRECTION.get(f.name, 0.0)
        if offset == 0.0:
            corrected.append(f)
        else:
            new_val = max(0.0, min(1.0, f.value + offset))
            corrected.append(Feature(
                dimension=f.dimension, name=f.name, value=new_val,
                intensity=f.intensity, confidence=f.confidence,
                usage_probability=f.usage_probability, stability=f.stability,
                evidence=f.evidence,
            ))
    return corrected
```

```python
def _detect_with_averaging(
    detector: Detector,
    conversation: str,
    profile_name: str,
    n_samples: int = 3,
) -> tuple[dict[str, float], dict[str, float]]:
    """Run detection n_samples times and return (means, stds)."""
    accumulated: dict[str, list[float]] = {}

    for i in range(n_samples):
        detected = detector.analyze(
            text=conversation,
            speaker_id=f"eval_{profile_name}_s{i}",
            speaker_label="Speaker",
        )
        for f in detected.features:
            key = f"{f.dimension}:{f.name}"
            accumulated.setdefault(key, []).append(f.value)

    means = {key: statistics.mean(vals) for key, vals in accumulated.items()}
    stds = {
        key: statistics.stdev(vals) if len(vals) > 1 else 0.0
        for key, vals in accumulated.items()
    }
    return means, stds
```

```python
def _spearman_rho(x: list[float], y: list[float]) -> float:
    """Compute Spearman rank correlation without scipy."""
    n = len(x)
    if n < 3:
        return float("nan")
    # Rank the values
    def _rank(vals):
        indexed = sorted(enumerate(vals), key=lambda t: t[1])
        ranks = [0.0] * n
        i = 0
        while i < n:
            j = i
            while j < n - 1 and indexed[j + 1][1] == indexed[j][1]:
                j += 1
            avg_rank = (i + j) / 2.0 + 1
            for k in range(i, j + 1):
                ranks[indexed[k][0]] = avg_rank
            i = j + 1
        return ranks
    rx, ry = _rank(x), _rank(y)
    d_sq = sum((a - b) ** 2 for a, b in zip(rx, ry))
    return 1 - (6 * d_sq) / (n * (n * n - 1))
```
